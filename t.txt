        // Function to update header based on login state
        function updateHeaderForLoginState() {
            const authButtons = document.querySelector('.auth-buttons');
            const userSection = document.querySelector('.user-section');
            
            // Check if user data exists in localStorage (you can use sessionStorage instead)
            const userData = localStorage.getItem('userData');
            
            if (userData) {
                // User is logged in
                const user = JSON.parse(userData);
                
                // Hide auth buttons
                authButtons.style.display = 'none';
                
                // Create welcome message
                const welcomeMessage = document.createElement('div');
                welcomeMessage.className = 'welcome-message';
                welcomeMessage.innerHTML = `

                // Insert welcome message after profile pic

                userSection.insertBefore(welcomeMessage, authButtons);
                
            } else {
                // User is not logged in - ensure auth buttons are visible
                authButtons.style.display = 'flex';
                
                // Remove any existing welcome message
                const welcomeMessage = document.querySelector('.welcome-message');
                if (welcomeMessage) {
                    welcomeMessage.remove();

        // Load cart from localStorage
function loadCart() {
    try {
        const savedCart = localStorage.getItem('shoppingCart');
        if (savedCart) {
            cart = JSON.parse(savedCart) || [];
            // Validate cart items
            cart = cart.filter(item => item && item.id);
        } else {
            cart = [];
        }
    } catch (error) {
        console.error('Error loading cart from localStorage:', error);
        cart = [];
    }
    updateCartCount();
    renderCartPreview();
}

        // Save cart to localStorage
        function saveCart() {
            localStorage.setItem('shoppingCart', JSON.stringify(cart));
        }

        // Add item to cart
        // Add item to cart
function addToCart(productId, productTitle, productPrice, productImage) {
    // Validate inputs
    if (!productId || !productTitle) {
        console.error('Invalid product data');
        return;
    }
    
    // Ensure price is a string with $ format
    let formattedPrice = productPrice;
    if (typeof productPrice === 'number') {
        formattedPrice = `$${productPrice.toFixed(2)}`;
    } else if (!productPrice.includes('$')) {
        formattedPrice = `$${productPrice}`;
    }
    
    // Find if product already exists in cart
    const existingItemIndex = cart.findIndex(item => item.id === productId);
    
    if (existingItemIndex > -1) {
        // Update quantity
        cart[existingItemIndex].quantity += 1;
    } else {
        // Add new item
        cart.push({
            id: productId,
            title: productTitle,
            price: formattedPrice,
            image: productImage,
            quantity: 1
        });
    }
    
    saveCart();
    updateCartCount();
    renderCartPreview();
    showAddedToCartMessage(productTitle);
    
    // Show cart preview automatically when adding item
    const cartPreview = document.querySelector('.cart-preview');
    if (cartPreview && !cartPreview.classList.contains('show')) {
        toggleCartPreview();
    }
}

        // Remove item from cart
        function removeFromCart(productId) {
            cart = cart.filter(item => item.id !== productId);
            saveCart();
            updateCartCount();
            renderCartPreview();
        }

        // Update item quantity
        function updateQuantity(productId, change) {
            const itemIndex = cart.findIndex(item => item.id === productId);
            
            if (itemIndex > -1) {
                cart[itemIndex].quantity += change;
                
                // Remove item if quantity is 0 or less
                if (cart[itemIndex].quantity <= 0) {
                    cart.splice(itemIndex, 1);
                }
                
                saveCart();
                updateCartCount();
                renderCartPreview();
            }
        }

        // Update cart count badge
        function updateCartCount() {
            const cartCountElement = document.querySelector('.cart-count');
            const totalItems = cart.reduce((total, item) => total + item.quantity, 0);
            
            if (cartCountElement) {
                cartCountElement.textContent = totalItems;
                
                // Add animation when count changes
                if (totalItems > 0) {
                    cartCountElement.style.display = 'flex';
                    cartCountElement.style.animation = 'none';
                    setTimeout(() => {
                        cartCountElement.style.animation = 'bounce 0.5s';
                    }, 10);
                } else {
                    cartCountElement.style.display = 'flex';
                }
            }
        }

        // Toggle cart preview visibility
        function toggleCartPreview() {
            const cartPreview = document.querySelector('.cart-preview');
            cartPreview.classList.toggle('show');
            
            // Render cart preview if showing
            if (cartPreview.classList.contains('show')) {
                renderCartPreview();
            }
        }

        // Render cart preview content
        // Render cart preview content
function renderCartPreview() {
    const cartItemsContainer = document.querySelector('.cart-items-container');
    const cartTotalElement = document.querySelector('.total-amount');
    
    if (!cart || cart.length === 0) {
        cartItemsContainer.innerHTML = `
            <div class="empty-cart-message">
                <i class="fas fa-shopping-basket"></i>
                <p>Your cart is empty</p>
            </div>
        `;
        if (cartTotalElement) cartTotalElement.textContent = '$0.00';
        return;
    }
    
    // Calculate total
    let total = 0;
    
    // Render cart items
    cartItemsContainer.innerHTML = cart.map(item => {
        if (!item || !item.price) {
            console.error('Invalid cart item:', item);
            return '';
        }
        
        // Safely extract price - handle different price formats
        let priceValue = 0;
        if (typeof item.price === 'string') {
            // Remove any non-numeric characters except decimal point
            priceValue = parseFloat(item.price.replace(/[^0-9.-]+/g, "")) || 0;
        } else if (typeof item.price === 'number') {
            priceValue = item.price;
        }
        
        const itemTotal = priceValue * (item.quantity || 1);
        total += itemTotal;
        
        return `
            <div class="cart-item" data-product-id="${item.id}">
                <div class="cart-item-image">
                    <img src="${item.image || ''}" alt="${item.title || 'Product'}">
                </div>
                <div class="cart-item-details">
                    <div class="cart-item-title">${item.title || 'Unknown Product'}</div>
                    <div class="cart-item-price">${item.price} Ã— ${item.quantity || 1}</div>
                    <div class="cart-item-actions">
                        <button class="quantity-btn" onclick="updateQuantity('${item.id}', -1)">-</button>
                        <span>${item.quantity || 1}</span>
                        <button class="quantity-btn" onclick="updateQuantity('${item.id}', 1)">+</button>
                        <button class="remove-item" onclick="removeFromCart('${item.id}')">Remove</button>
                    </div>
                </div>
            </div>
        `;
    }).join('');
    
    // Update total
    if (cartTotalElement) {
        cartTotalElement.textContent = `$${total.toFixed(2)}`;
    }
}
